def check_timestamp_irregularities(timestamps):
    # Calculate differences between consecutive timestamps
    time_diffs = np.diff(timestamps)

    # Check if the differences are consistent (e.g., within a small tolerance)
    tolerance = 0.1 * np.mean(time_diffs)  # 10% of the mean difference
    irregularities = np.abs(time_diffs - np.mean(time_diffs)) > tolerance

    if np.any(irregularities):
        # print(f"Irregularities found in timestamps: {time_diffs[irregularities]}")
        return True
    else:
        # print("No irregularities found in timestamps.")
        return False

# 1.1 Gravity Compensation
from scipy.signal import butter, filtfilt

# remove low-frequency components (like gravity) from an accelerometer signal  using a high-pass filter.
# cutoff=0.5: Because human motion (tremor, posture, walking) typically happens above 0.5 Hz
def remove_gravity(accel_series, fs, cutoff=0.5):
    
    if len(accel_series) < 16:
        # print(f"Skipping gravity removal: only {len(accel_series)} samples")
        return accel_series

    # Filter order=4: Controls steepness of the cutoff
    # cutoff/fs/2: Normalized cutoff for filter    
    # 'highpass': Remove frequencies below cutoff Hz, keeps only motion-related signal
    # b, a are filter coefficients (arrays), used for filtering
    b, a = butter(4, cutoff/(fs/2), 'highpass')
    
    # applies the filter forward and backward to prevent phase shift
    # Filters out slow-changing components like gravity (0 Hz).
    # Keeps higher-frequency components  usually real motion (e.g., hand tremor)
    return filtfilt(b, a, accel_series)

# calculate window samples for each set of signal
# desired_time_ms = 250 to retain full tremor details
def get_rolling_window(signal, fs, desired_time_ms=250, min_window=3):
    N = len(signal)
    desired_time_sec = desired_time_ms / 1000
    win = int(fs * desired_time_sec)
    return min(N, max(min_window, win))

def denoise(data, window):
    # Remove spike by median value of range window items where the spike is the middle value
    # return data.rolling(window=window, center=True).median()

def calculate_smv(df):
    return np.sqrt(df['Accel_X_clean']**2 + df['Accel_Y_clean']**2 + df['Accel_Z_clean']**2)

def compute_smv(input_dic):
    for participant_id, participant_data in input_dic.items():
        for index, row in participant_data['activity'].iterrows():
            df_sensor = pd.DataFrame()
            for sensor in row['sensor data']: 
                df_sensor = sensor['sensor data']
                df_sensor['SMV'] = calculate_smv(df_sensor)
    return input_dic

import numpy as np
from scipy.signal import welch
from scipy.fft import fft, fftfreq
import antropy as ant

# This gives how much energy lies in the tremor frequency band.
def compute_relative_power(signal, fs, band=(3, 10)):
    try:
        if len(signal) < 16:
            return 0.0  # fallback for tiny windows
        f, Pxx = welch(signal, fs=fs, nperseg=min(64, len(signal)))
        total_power = np.sum(Pxx)
        band_power = np.sum(Pxx[(f >= band[0]) & (f <= band[1])])
        return band_power / total_power if total_power > 0 else 0.0
    except:
        return 0.0

# This counts how often the signal crosses zero, calculated based on product of two consecutive values is negative
def compute_zero_crossing_rate(signal):
    signal = np.asarray(signal)  # Convert list to NumPy array
    # return ((signal[:-1] * signal[1:]) < 0).sum() / max(len(signal), 1)
    return ((signal[:-1] * signal[1:]) < 0).sum() / max(len(signal) - 1, 1)

# Measures complexity/irregularity of a signal.
def compute_sample_entropy(signal):
    try:
        if np.isnan(signal).any() or np.isinf(signal).any():
            raise ValueError("Signal contains NaN or inf values.")
        entropy = ant.sample_entropy(signal) if len(signal) >= 10 else 0.0
        if np.isnan(entropy) or np.isinf(entropy):
            # print("Entropy is NaN or inf, returning 0.0")
            # print("len of signal:", len(signal))
            return 0.0
        return entropy
    except:
        return 0.0

# dominant_freq: dominant frequency in the 310 Hz range.
# dominant_power: the strength/amplitude of that frequency component
# fs : samples per second
def compute_fft_peak(signal, fs):
    
    # Convert to NumPy array
    signal = np.asarray(signal)
    if len(signal) < 8:
        return 0.0, 0.0  # fallback

    yf = np.abs(fft(signal))
    
    # Get frequency values
    xf = fftfreq(len(signal), 1/fs)
    
    # Keep only positive frequencies     
    pos_mask = xf > 0
    xf = xf[pos_mask]
    yf = yf[pos_mask]
    
    # Keep 310 Hz (tremor band)
    tremor_mask = (xf >= 3) & (xf <= 10)

    if not np.any(tremor_mask):
        return 0.0, 0.0

    dominant_freq = xf[tremor_mask][np.argmax(yf[tremor_mask])]
    dominant_power = np.max(yf[tremor_mask])
    return dominant_freq, dominant_power

def extract_features(signal, fs):
    centered = signal - np.mean(signal)
    freq, power = compute_fft_peak(centered, fs)
    
    return {
        'mean': np.mean(centered),
        'std': np.std(centered),
        'ptp': np.ptp(centered),
        'zcr': compute_zero_crossing_rate(centered),
        'rel_power': compute_relative_power(centered, fs),
        'sample_entropy': compute_sample_entropy(centered),
        'tremor_freq': freq,
        'tremor_power': power
    }

def remove_outliers_modified_zscore(df, columns, threshold=3.5):
    mask = pd.Series(True, index=df.index)
    for col in columns:
        median = df[col].median()
        mad = np.median(np.abs(df[col] - median))
        if mad == 0:
            continue
        modified_z = 0.6745 * (df[col] - median) / mad
        mask &= np.abs(modified_z) < threshold
    return df[mask]

# Perform statistical tests to check for significant differences between groups regarding the features
from scipy.stats import ttest_ind, mannwhitneyu, shapiro
import pandas as pd

# Split data into tremor and no tremor
group_0 = df_no_outliers[df_no_outliers[target_col] == 0.0]
group_1 = df_no_outliers[df_no_outliers[target_col] == 1.0]

# let check tremor_std and tremor_ptp to see if we need to remove them from the dataset.
features = ['tremor_std', 'tremor_ptp']

results = []

for feature in features:
    data0 = group_0[feature].dropna()
    data1 = group_1[feature].dropna()

    # Test for normality
    stat0, p0 = shapiro(data0)
    stat1, p1 = shapiro(data1)

    normal = p0 > 0.05 and p1 > 0.05

    if normal:
        stat, p_value = ttest_ind(data0, data1)
        test_type = 't-test'
    else:
        stat, p_value = mannwhitneyu(data0, data1)
        test_type = 'Mann-Whitney U'

    results.append({
        'feature': feature,
        'test_used': test_type,
        'p_value': p_value
    })

# Show results in DataFrame
results_df = pd.DataFrame(results)
print(results_df.sort_values(by='p_value'))

      feature       test_used   p_value
0  tremor_std  Mann-Whitney U  0.031094
1  tremor_ptp  Mann-Whitney U  0.131271
  • p_value of tremor_std < 0.05 means that this feature shows significant difference between the 2 groups so it should be kept
  • p_value of tremore_ptp > 0.05 so we should remove this feature from the dataset

def stratified_split_by_participant_fixed(df, target_col, participant_col, test_size=0.2, random_state=42):

    # Aggregate target column by participant
    participant_target = df.groupby(participant_col)[target_col].apply(lambda x: 1 if (x == 1).any() else 0).reset_index()
    participant_target.rename(columns={target_col: 'has_target'}, inplace=True)

    # Stratified split on participants
    train_participants, test_participants = train_test_split(
        participant_target[participant_col],
        test_size=test_size,
        stratify=participant_target['has_target'],
        random_state=random_state
    )

    # Filter the original dataset based on the split
    train_data = df[df[participant_col].isin(train_participants)]
    test_data = df[df[participant_col].isin(test_participants)]

    # Verify proportions
    print("Train set proportions and count:")
    print(train_data[target_col].value_counts(normalize=True), train_data[target_col].value_counts())

    print("Test set proportions and count:")
    print(test_data[target_col].value_counts(normalize=True), test_data[target_col].value_counts())
    
    return train_data, test_data

def plot_roc_curve(y_true, y_scores):
    from sklearn.metrics import roc_curve, auc
    fpr, tpr, thresholds = roc_curve(y_true, y_scores)
    roc_auc = auc(fpr, tpr)

    plt.figure(figsize=(8, 6))
    plt.plot(fpr, tpr, color='blue', lw=2, label=f'ROC curve (area = {roc_auc:.2f})')
    plt.plot([0, 1], [0, 1], color='red', lw=2, linestyle='--')
    plt.xlim([0.0, 1.0])
    plt.ylim([0.0, 1.05])
    plt.xlabel('False Positive Rate')
    plt.ylabel('True Positive Rate')
    plt.title('Receiver Operating Characteristic (ROC) Curve')
    plt.legend(loc='lower right')
    plt.grid()
    plt.show()

        # Raw splits
        X_train_raw, X_val_raw = X_train.iloc[train_idx], X_train.iloc[val_idx]
        y_train_raw, y_val_raw = y_train.iloc[train_idx], y_train.iloc[val_idx]

        # Fit scaler only on training data
        scaler = StandardScaler()
        X_train_scaled = scaler.fit_transform(X_train_raw)
        X_test_scaled = scaler.transform(X_val_raw)  # Use same scaler

        # Train model
        model = LogisticRegression(
            # multi_class='ovr',

            class_weight='balanced',  # helpful for imbalanced classes
